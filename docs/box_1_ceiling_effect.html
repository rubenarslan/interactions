Ceiling Effect Simulation

Generate data

set.seed(12345)
library(tidyverse)
n <- 500
ceiling <- tibble(
  Moderator = c(rep(0, n/5*4), rep(1, n/5)),
  pred = 1.4 * Moderator + 0.7 * rnorm(n),
  out = 1.1 * Moderator + 0.6 * pred + 0.5 * rnorm(n)
) %>% 
  mutate(
  Moderator = as.factor(Moderator)
  )
sd(ceiling$out)
## [1] 0.9513978

ceiling$censored <- "none"
